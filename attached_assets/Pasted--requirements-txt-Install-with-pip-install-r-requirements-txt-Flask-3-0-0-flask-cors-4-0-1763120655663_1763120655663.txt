import os
from datetime import timedelta

class Config:
    """Base configuration"""
    SECRET_KEY = os.environ.get('SECRET_KEY') or 'dev-secret-key-change-in-production'
    
    # API Rate Limits
    RATE_LIMIT_ENABLED = True
    RATE_LIMIT_REQUESTS = 100  # requests per minute
    
    # Data Update Configuration
    UPDATE_INTERVAL = 3600  # 1 hour in seconds
    
    # Cache Configuration
    CACHE_ENABLED = True
    CACHE_TTL = 1800  # 30 minutes
    
    # VirusTotal Configuration
    VIRUSTOTAL_RATE_LIMIT = 4  # requests per minute (free tier)
    VIRUSTOTAL_REQUEST_DELAY = 15  # seconds between requests
    
    # NVD Configuration
    NVD_DEFAULT_DAYS = 7  # Default days to look back for CVEs
    
    # Logging
    LOG_LEVEL = 'INFO'
    LOG_FILE = 'threat_intel.log'


class DevelopmentConfig(Config):
    """Development configuration"""
    DEBUG = True
    TESTING = False


class ProductionConfig(Config):
    """Production configuration"""
    DEBUG = False
    TESTING = False
    
    # In production, use environment variables
    SECRET_KEY = os.environ.get('SECRET_KEY')
    
    # Stricter rate limits in production
    RATE_LIMIT_REQUESTS = 50


class TestingConfig(Config):
    """Testing configuration"""
    DEBUG = True
    TESTING = True
    
    # Faster updates for testing
    UPDATE_INTERVAL = 300  # 5 minutes


# Configuration dictionary
config = {
    'development': DevelopmentConfig,
    'production': ProductionConfig,
    'testing': TestingConfig,
    'default': DevelopmentConfig
}
